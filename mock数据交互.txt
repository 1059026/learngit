
模拟接口：安装mock.js : npm install mockjs


创建src/api/mock.js文件
import Mock from 'mockjs'


Mock.mock('/list',{
'yule':[{
title:'娱乐标题1',
content:'内容1'
}
],
'shehui':[{
title:'社会标题1',
content:'内容1'
},
{
title:'社会标题2',
content:'内容2'
}
]


main.js文件引入：import './api/mock.js'


ListView.vue文件写请求
mounted (){
let id = this.$route.param.id;
this.$http({
method:'get',
url: '/list'
}).then(res => {
this.arr = res.data[id];
})
}


在main.js文件中引入axios
import axios from 'axios' //console.log(axios) 是个函数 
Vue.prototype.$http = axios //Vue.prototype:相当于给vue挂载一个方法 console.log(thia.$http)
